#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <bpf/bpf_helpers.h>

#define MAX_ENTRIES 65536
#define RATE_LIMIT_PPS 1000
#define GLOBAL_RATE_LIMIT_PPS 500000
#define WINDOW_NS 1000000000ULL

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 10240);
    __type(key, __u32);
    __type(value, __u32);
} blacklist_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_LRU_HASH);
    __uint(max_entries, MAX_ENTRIES);
    __type(key, __u32);
    __type(value, __u32);
} meter_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 1024);
    __type(key, __u32);
    __type(value, __u32);
} allowlist_map SEC(".maps");

struct token_bucket {
    __u64 last_time;
    __u32 tokens;
};

struct {
    __uint(type, BPF_MAP_TYPE_LRU_HASH);
    __uint(max_entries, MAX_ENTRIES);
    __type(key, __u32);
    __type(value, struct token_bucket);
} rate_map SEC(".maps");

struct global_rate {
    __u64 window_start;
    __u32 count;
};

struct {
    __uint(type, BPF_MAP_TYPE_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, struct global_rate);
} global_pps_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 256);
    __type(key, __u8);
    __type(value, __u32);
} proto_filter_map SEC(".maps");

SEC("xdp")
int fluxguard_filter(struct xdp_md *ctx) {
    void *data_end = (void *)(long)ctx->data_end;
    void *data = (void *)(long)ctx->data;

    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end)
        return XDP_PASS;

    if (eth->h_proto != __constant_htons(ETH_P_IP))
        return XDP_PASS;

    struct iphdr *iph = data + sizeof(struct ethhdr);
    if ((void *)(iph + 1) > data_end)
        return XDP_PASS;

    __u32 src_ip = iph->saddr;
    __u8 protocol = iph->protocol;

    /* 1. Allowlist */
    __u32 *allowed = bpf_map_lookup_elem(&allowlist_map, &src_ip);
    if (allowed)
        return XDP_PASS;

    /* 2. Global shields-up mode */
    __u32 global_key = 0;
    struct global_rate *gr = bpf_map_lookup_elem(&global_pps_map, &global_key);
    __u64 now = bpf_ktime_get_ns();
    if (gr) {
        if (now - gr->window_start > WINDOW_NS) {
            gr->window_start = now;
            gr->count = 0;
        }
        __sync_fetch_and_add(&gr->count, 1);
        if (gr->count > GLOBAL_RATE_LIMIT_PPS) {
            return XDP_DROP;
        }
    }

    /* 3. Protocol filter check */
    __u32 *proto_action = bpf_map_lookup_elem(&proto_filter_map, &protocol);
    if (proto_action && *proto_action == 1) {
        return XDP_DROP;
    }

    /* 4. Update meter_map */
    __u32 *cnt = bpf_map_lookup_elem(&meter_map, &src_ip);
    if (cnt) {
        __sync_fetch_and_add(cnt, 1);
    } else {
        __u32 init = 1;
        bpf_map_update_elem(&meter_map, &src_ip, &init, BPF_ANY);
    }

    /* 5. Check blacklist */
    __u32 *blocked = bpf_map_lookup_elem(&blacklist_map, &src_ip);
    if (blocked)
        return XDP_DROP;

    /* 6. In-kernel token bucket */
    struct token_bucket *tb = bpf_map_lookup_elem(&rate_map, &src_ip);
    if (tb) {
        __u64 elapsed = now - tb->last_time;
        if (elapsed > WINDOW_NS) {
            tb->last_time = now;
            tb->tokens = 1;
        } else {
            __sync_fetch_and_add(&tb->tokens, 1);
            if (tb->tokens > RATE_LIMIT_PPS) {
                return XDP_DROP;
            }
        }
    } else {
        struct token_bucket new_tb = {0};
        new_tb.last_time = now;
        new_tb.tokens = 1;
        bpf_map_update_elem(&rate_map, &src_ip, &new_tb, BPF_ANY);
    }

    return XDP_PASS;
}

char _license[] SEC("license") = "GPL";

#!/usr/bin/env python3
from __future__ import annotations
import argparse
import ctypes
import datetime
import ipaddress
import json
import os
import signal
import sys
import threading
import time
from typing import Any, Dict, List, Optional, Tuple, TextIO

from fluxguard_metrics import MetricsState, start_metrics_server

libc = ctypes.CDLL("libc.so.6", use_errno=True)

class bpf_attr(ctypes.Union):
    class map_elem(ctypes.Structure):
        _fields_ = [
            ("map_fd", ctypes.c_uint32),
            ("key", ctypes.c_uint64),
            ("value", ctypes.c_uint64),
            ("flags", ctypes.c_uint64),
        ]
    class obj_get(ctypes.Structure):
        _fields_ = [
            ("pathname", ctypes.c_uint64),
            ("bpf_fd", ctypes.c_uint32),
            ("file_flags", ctypes.c_uint32),
        ]
    _fields_ = [("elem", map_elem), ("obj_get", obj_get)]

def bpf_syscall(cmd: int, attr: bpf_attr) -> int:
    res = libc.syscall(321, cmd, ctypes.byref(attr), ctypes.sizeof(attr))
    if res < 0:
        err = ctypes.get_errno()
        raise OSError(err, os.strerror(err))
    return res

def bpf_obj_get(path: str) -> int:
    attr = bpf_attr()
    attr.obj_get.pathname = ctypes.cast(ctypes.c_char_p(path.encode()), ctypes.c_void_p).value
    return bpf_syscall(7, attr)

class BpfKey(ctypes.Structure):
    _fields_ = [("bytes", ctypes.c_uint8 * 4)]

class BpfValue(ctypes.Structure):
    _fields_ = [("val", ctypes.c_uint32)]

class ProtoKey(ctypes.Structure):
    _fields_ = [("val", ctypes.c_uint8)]

def key_to_ip(k: BpfKey) -> str:
    return str(ipaddress.IPv4Address(bytes(k.bytes)))

def ip_to_key(ip_str: str) -> BpfKey:
    k = BpfKey()
    k.bytes = (ctypes.c_uint8 * 4)(*ipaddress.IPv4Address(ip_str).packed)
    return k

def dump_map_u32(fd: int) -> Dict[str, int]:
    res = {}
    prev_key = BpfKey()
    next_key = BpfKey()
    value = BpfValue()
    
    attr = bpf_attr()
    attr.elem.map_fd = fd
    attr.elem.key = 0
    attr.elem.value = ctypes.addressof(next_key)
    
    try:
        bpf_syscall(4, attr)
    except OSError as e:
        if e.errno == 2:
            return res
        raise
        
    while True:
        attr_lookup = bpf_attr()
        attr_lookup.elem.map_fd = fd
        attr_lookup.elem.key = ctypes.addressof(next_key)
        attr_lookup.elem.value = ctypes.addressof(value)
        try:
            bpf_syscall(1, attr_lookup)
            res[key_to_ip(next_key)] = value.val
        except OSError:
            pass
            
        ctypes.memmove(ctypes.addressof(prev_key), ctypes.addressof(next_key), ctypes.sizeof(BpfKey))
        attr_next = bpf_attr()
        attr_next.elem.map_fd = fd
        attr_next.elem.key = ctypes.addressof(prev_key)
        attr_next.elem.value = ctypes.addressof(next_key)
        try:
            bpf_syscall(4, attr_next)
        except OSError as e:
            if e.errno == 2:
                break
            attr_next.elem.key = 0
            try:
                bpf_syscall(4, attr_next)
                continue
            except OSError:
                break
    return res

def set_map_u32(fd: int, ip_str: str, val: int) -> None:
    k = ip_to_key(ip_str)
    v = BpfValue()
    v.val = val
    attr = bpf_attr()
    attr.elem.map_fd = fd
    attr.elem.key = ctypes.addressof(k)
    attr.elem.value = ctypes.addressof(v)
    attr.elem.flags = 0
    bpf_syscall(2, attr)

def del_map_u32(fd: int, ip_str: str) -> None:
    k = ip_to_key(ip_str)
    attr = bpf_attr()
    attr.elem.map_fd = fd
    attr.elem.key = ctypes.addressof(k)
    bpf_syscall(3, attr)

def set_proto_filter(fd: int, proto: int, action: int) -> None:
    k = ProtoKey()
    k.val = proto
    v = BpfValue()
    v.val = action
    attr = bpf_attr()
    attr.elem.map_fd = fd
    attr.elem.key = ctypes.addressof(k)
    attr.elem.value = ctypes.addressof(v)
    attr.elem.flags = 0
    bpf_syscall(2, attr)

shutdown_event = threading.Event()

def handle_sigterm(signum: int, frame: Any) -> None:
    shutdown_event.set()

def write_json_log(fd: TextIO, event: str, fields: Dict[str, Any]) -> None:
    ts = datetime.datetime.now(datetime.timezone.utc).strftime("%Y-%m-%dT%H:%M:%S.%f")[:-3] + "Z"
    record: Dict[str, Any] = {"timestamp": ts, "event": event}
    record.update(fields)
    line = json.dumps(record, separators=(",", ":")) + "\n"
    fd.write(line)
    fd.flush()

def monitor(args: argparse.Namespace, metrics: MetricsState, log_fd: TextIO) -> None:
    base = f"/sys/fs/bpf/{args.netns}"
    meter_fd = bpf_obj_get(f"{base}/meter_map")
    blacklist_fd = bpf_obj_get(f"{base}/blacklist_map")
    allowlist_fd = bpf_obj_get(f"{base}/allowlist_map")
    proto_fd = bpf_obj_get(f"{base}/proto_filter_map")

    if args.drop_proto is not None:
        set_proto_filter(proto_fd, args.drop_proto, 1)
        print(f"Set protocol filter rule: drop protocol {args.drop_proto}")

    prev_counts: Dict[str, int] = {}
    prev_ts = time.monotonic()
    blocked_until: Dict[str, float] = {}
    allowlist_cache: Dict[str, int] = {}
    allowlist_ts = 0.0

    print(f"Phase 7 Brain (ctypes) started. threshold={args.pps_threshold:.1f} pps")
    
    while not shutdown_event.is_set():
        now = time.monotonic()
        dt = now - prev_ts
        if dt <= 0:
            dt = args.poll_interval

        if now - allowlist_ts >= 30.0:
            try:
                allowlist_cache = dump_map_u32(allowlist_fd)
                allowlist_ts = now
            except OSError:
                pass

        try:
            curr_counts = dump_map_u32(meter_fd)
        except OSError as exc:
            time.sleep(args.poll_interval)
            prev_ts = time.monotonic()
            continue

        to_block = []
        for ip, curr in curr_counts.items():
            if ip in allowlist_cache:
                continue
            prev = prev_counts.get(ip, curr)
            pps = (curr - prev if curr >= prev else curr) / dt
            if pps >= args.pps_threshold and ip not in blocked_until:
                to_block.append((ip, pps, curr))

        if to_block:
            to_block.sort(key=lambda x: x[1], reverse=True)
            if args.block_batch_size > 0:
                to_block = to_block[: args.block_batch_size]

            for ip, pps, counter in to_block:
                try:
                    set_map_u32(blacklist_fd, ip, 1)
                    blocked_until[ip] = now + args.cooldown_sec
                    metrics.inc_block_events()
                    print(f"[BLOCK] ip={ip} pps={pps:.1f} counter={counter} unblock_in={args.cooldown_sec}s")
                    write_json_log(log_fd, "BLOCK", {"ip": ip, "pps": round(pps, 6), "counter": counter, "blocked_count": len(blocked_until)})
                except OSError:
                    pass

        if blocked_until:
            expired = [ip for ip, ts in blocked_until.items() if now >= ts]
            for ip in expired:
                try:
                    del_map_u32(blacklist_fd, ip)
                    metrics.inc_unblock_events()
                    print(f"[UNBLOCK] ip={ip} cooldown_expired")
                    write_json_log(log_fd, "UNBLOCK", {"ip": ip, "blocked_count": len(blocked_until) - 1})
                except OSError:
                    pass
                finally:
                    blocked_until.pop(ip, None)

        active_blocked = len(blocked_until)
        if not args.no_log_ticks:
            write_json_log(log_fd, "TICK", {"observed_ips": len(curr_counts), "blocked_count": active_blocked})
        metrics.update_snapshot(curr_counts, active_blocked)

        if args.verbose:
            print(f"[TICK] ips={len(curr_counts)} blocked={active_blocked} dt={dt:.3f}s")

        prev_counts = curr_counts
        prev_ts = now
        
        sleep_left = min(args.poll_interval, 1.0)
        while sleep_left > 0 and not shutdown_event.is_set():
            chunk = min(0.1, sleep_left)
            time.sleep(chunk)
            sleep_left -= chunk

def main() -> int:
    signal.signal(signal.SIGTERM, handle_sigterm)
    signal.signal(signal.SIGINT, handle_sigterm)

    p = argparse.ArgumentParser()
    p.add_argument("--netns", default="fluxguard")
    p.add_argument("--pps-threshold", type=float, default=10000.0)
    p.add_argument("--poll-interval", type=float, default=0.2)
    p.add_argument("--cooldown-sec", type=int, default=900)
    p.add_argument("--block-batch-size", type=int, default=256)
    p.add_argument("--verbose", action="store_true")
    p.add_argument("--no-log-ticks", action="store_true")
    p.add_argument("--log-file", default="/home/samarth/fluxguard/fluxguard.log")
    p.add_argument("--metrics-host", default="127.0.0.1")
    p.add_argument("--metrics-port", type=int, default=9090)
    p.add_argument("--drop-proto", type=int, help="Protocol ID to drop")
    args = p.parse_args()

    metrics = MetricsState()
    start_metrics_server(metrics, args.metrics_host, args.metrics_port)

    with open(args.log_file, "a", encoding="utf-8") as log_fd:
        try:
            monitor(args, metrics, log_fd)
        except Exception as exc:
            print(f"[FATAL] {exc}", file=sys.stderr)
            return 1
            
    print("\nFluxGuard brain shutdown complete.")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
