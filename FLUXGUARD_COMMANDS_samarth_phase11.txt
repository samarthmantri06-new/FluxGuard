# --- Step 2: Mount BPF filesystem (if not already mounted) ---
sudo mount -t bpf none /sys/fs/bpf/ 2>/dev/null || true
sudo ip netns exec fluxguard mount -t bpf none /sys/fs/bpf/ 2>/dev/null || true
sudo mkdir -p /sys/fs/bpf/fluxguard

# --- Step 3: Compile Phase 11 XDP kernel program ---
cd /home/samarth/fluxguard
clang -O2 -g -Wall -target bpf \
    -I/usr/include/$(uname -m)-linux-gnu \
    -c fluxguard_kern.c -o fluxguard_kern.o

# Verify the object loaded and inspect its sections
llvm-objdump -h fluxguard_kern.o | grep -E "maps|xdp"

# --- Step 4: Reload XDP program (generic mode for VirtualBox VMs) ---
sudo ip netns exec fluxguard ip link set dev veth-fg-in xdp off 2>/dev/null || true
sudo rm -rf /sys/fs/bpf/fluxguard/*

# Try native driver mode first, fall back to generic (VirtualBox always uses generic)
if sudo ip netns exec fluxguard ip link set dev veth-fg-in xdpdrv obj fluxguard_kern.o sec xdp 2>/dev/null; then
    echo "[INFO] XDP loaded in NATIVE (xdpdrv) mode"
else
    echo "[INFO] Falling back to GENERIC (xdpgeneric) mode — normal for VirtualBox"
    sudo ip netns exec fluxguard ip link set dev veth-fg-in xdpgeneric obj fluxguard_kern.o sec xdp
fi

# Confirm XDP is attached
sudo ip netns exec fluxguard ip link show veth-fg-in | grep -i xdp

# --- Step 5: Pin ALL maps (IPv4 + IPv6 + shared) ---
# Phase 11 has 11 maps total — pin each one
for map in \
    meter_map blacklist_map allowlist_map rate_map \
    meter_map_v6 blacklist_map_v6 allowlist_map_v6 rate_map_v6 \
    global_counter_map global_rate_map proto_filter_map event_ringbuf; do

    ID=$(sudo ip netns exec fluxguard bpftool map show | grep -w "$map" | head -1 | awk '{print $1}' | tr -d ':')
    if [ -n "$ID" ]; then
        sudo ip netns exec fluxguard bpftool map pin id "$ID" /sys/fs/bpf/fluxguard/"$map"
        echo "[PIN] $map -> /sys/fs/bpf/fluxguard/$map (id=$ID)"
    else
        echo "[WARN] Map not found: $map"
    fi
done

# Verify all maps are pinned
ls -la /sys/fs/bpf/fluxguard/

# --- Step 6: Verify kernel architecture and BPF syscall ID ---
uname -m
# Expected: x86_64 -> SYS_BPF=321 (auto-detected by fluxguard_bpf.py)

# Verify CPU count (affects PERCPU_ARRAY read buffer sizing in fluxguard_bpf.py)
nproc
# The brain will use os.cpu_count() dynamically — no hardcoding
