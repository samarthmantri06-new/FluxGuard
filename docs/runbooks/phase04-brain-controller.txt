#!/usr/bin/env python3
from __future__ import annotations

import argparse
import ipaddress
import json
import subprocess
import sys
import time
from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple


@dataclass
class Config:
    netns: str
    meter_map: str
    blacklist_map: str
    pps_threshold: float
    poll_interval: float
    cooldown_sec: int
    block_batch_size: int
    verbose: bool


def run_cmd(cmd: List[str]) -> str:
    try:
        proc = subprocess.run(
            cmd,
            check=True,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )
    except subprocess.CalledProcessError as exc:
        raise RuntimeError(
            f"Command failed: {' '.join(cmd)}\n"
            f"exit={exc.returncode}\n"
            f"stdout={exc.stdout.strip()}\n"
            f"stderr={exc.stderr.strip()}"
        ) from exc
    return proc.stdout


def bpftool_json(netns: str, args: List[str]) -> object:
    cmd = ["sudo", "ip", "netns", "exec", netns, "bpftool", "--json"] + args
    out = run_cmd(cmd)
    try:
        return json.loads(out)
    except json.JSONDecodeError as exc:
        raise RuntimeError(f"Invalid JSON from bpftool:\n{out}") from exc


def ip_to_key_hex(ip_str: str) -> str:
    ip = ipaddress.IPv4Address(ip_str)
    return " ".join(f"{b:02x}" for b in ip.packed)


def extract_hex_tokens(field: object) -> List[str]:
    tokens: List[str] = []
    if isinstance(field, list):
        for x in field:
            if isinstance(x, int):
                tokens.append(f"{x & 0xFF:02x}")
            elif isinstance(x, str):
                s = x.strip().lower().replace("0x", "")
                if s:
                    tokens.extend(s.split())
    elif isinstance(field, str):
        s = field.strip().lower().replace("0x", "")
        if s:
            tokens.extend(s.split())
    elif isinstance(field, int):
        tokens.append(f"{field & 0xFF:02x}")
    elif isinstance(field, dict):
        for k in ("bytes", "data", "raw"):
            if k in field:
                return extract_hex_tokens(field[k])
    return tokens


def key_to_ipv4(tokens: List[str]) -> Optional[str]:
    if len(tokens) < 4:
        return None
    try:
        b = bytes(int(t, 16) for t in tokens[:4])
        return str(ipaddress.IPv4Address(b))
    except Exception:
        return None


def value_to_u32(tokens: List[str]) -> Optional[int]:
    if len(tokens) < 4:
        return None
    try:
        b0, b1, b2, b3 = (int(tokens[i], 16) for i in range(4))
        return b0 | (b1 << 8) | (b2 << 16) | (b3 << 24)
    except Exception:
        return None


def parse_meter(raw: object) -> Dict[str, int]:
    res: Dict[str, int] = {}
    if not isinstance(raw, list):
        return res
    for e in raw:
        if not isinstance(e, dict):
            continue
        k = e.get("key")
        v = e.get("value")
        if k is None or v is None:
            continue
        ip = key_to_ipv4(extract_hex_tokens(k))
        cnt = value_to_u32(extract_hex_tokens(v))
        if ip is None or cnt is None:
            continue
        res[ip] = cnt
    return res


def dump_meter(cfg: Config) -> Dict[str, int]:
    return parse_meter(bpftool_json(cfg.netns, ["map", "dump", "name", cfg.meter_map]))


def blacklist_update(cfg: Config, ip_str: str) -> None:
    key_hex = ip_to_key_hex(ip_str)
    cmd = [
        "sudo", "ip", "netns", "exec", cfg.netns,
        "bpftool", "map", "update", "name", cfg.blacklist_map,
        "key", "hex",
    ] + key_hex.split() + ["value", "hex", "01", "00", "00", "00"]
    run_cmd(cmd)


def blacklist_delete(cfg: Config, ip_str: str) -> None:
    key_hex = ip_to_key_hex(ip_str)
    cmd = [
        "sudo", "ip", "netns", "exec", cfg.netns,
        "bpftool", "map", "delete", "name", cfg.blacklist_map,
        "key", "hex",
    ] + key_hex.split()
    run_cmd(cmd)


def monitor(cfg: Config) -> None:
    prev: Dict[str, int] = {}
    prev_t = time.monotonic()
    blocked_until: Dict[str, float] = {}

    print(f"brain started threshold={cfg.pps_threshold}pps cooldown={cfg.cooldown_sec}s")

    while True:
        now = time.monotonic()
        dt = now - prev_t
        if dt <= 0:
            dt = cfg.poll_interval

        cur = dump_meter(cfg)

        to_block: List[Tuple[str, float, int]] = []
        for ip, c in cur.items():
            p = prev.get(ip, c)
            d = c - p if c >= p else c
            pps = d / dt
            if pps >= cfg.pps_threshold and ip not in blocked_until:
                to_block.append((ip, pps, c))

        if to_block:
            to_block.sort(key=lambda x: x[1], reverse=True)
            if cfg.block_batch_size > 0:
                to_block = to_block[: cfg.block_batch_size]
            for ip, pps, c in to_block:
                blacklist_update(cfg, ip)
                blocked_until[ip] = now + cfg.cooldown_sec
                print(f"BLOCK ip={ip} pps={pps:.1f} counter={c}")

        if blocked_until:
            expired = [ip for ip, t in blocked_until.items() if now >= t]
            for ip in expired:
                try:
                    blacklist_delete(cfg, ip)
                except Exception:
                    pass
                blocked_until.pop(ip, None)
                print(f"UNBLOCK ip={ip}")

        if cfg.verbose:
            print(f"TICK ips={len(cur)} blocked={len(blocked_until)} dt={dt:.3f}s")

        prev = cur
        prev_t = now
        time.sleep(cfg.poll_interval)


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--netns", default="fluxguard")
    ap.add_argument("--meter-map", default="meter_map")
    ap.add_argument("--blacklist-map", default="blacklist_map")
    ap.add_argument("--pps-threshold", type=float, default=1000.0)
    ap.add_argument("--poll-interval", type=float, default=0.2)
    ap.add_argument("--cooldown-sec", type=int, default=30)
    ap.add_argument("--block-batch-size", type=int, default=256)
    ap.add_argument("--verbose", action="store_true")
    a = ap.parse_args()

    cfg = Config(
        netns=a.netns,
        meter_map=a.meter_map,
        blacklist_map=a.blacklist_map,
        pps_threshold=a.pps_threshold,
        poll_interval=a.poll_interval,
        cooldown_sec=a.cooldown_sec,
        block_batch_size=a.block_batch_size,
        verbose=a.verbose,
    )
    monitor(cfg)
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <bpf/bpf_helpers.h>

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 10240);
    __type(key, __u32);
    __type(value, __u32);
} blacklist_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 65536);
    __type(key, __u32);
    __type(value, __u32);
} meter_map SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 1024);
    __type(key, __u32);
    __type(value, __u32);
} allowlist_map SEC(".maps");

SEC("xdp")
int fluxguard_filter(struct xdp_md *ctx) {
    void *data_end = (void *)(long)ctx->data_end;
    void *data = (void *)(long)ctx->data;

    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end)
        return XDP_PASS;

    if (eth->h_proto != __constant_htons(ETH_P_IP))
        return XDP_PASS;

    struct iphdr *iph = data + sizeof(struct ethhdr);
    if ((void *)(iph + 1) > data_end)
        return XDP_PASS;

    __u32 src_ip = iph->saddr;

    __u32 *allowed = bpf_map_lookup_elem(&allowlist_map, &src_ip);
    if (allowed)
        return XDP_PASS;

    __u32 *cnt = bpf_map_lookup_elem(&meter_map, &src_ip);
    if (cnt) {
        __sync_fetch_and_add(cnt, 1);
    } else {
        __u32 init = 1;
        bpf_map_update_elem(&meter_map, &src_ip, &init, BPF_ANY);
    }

    __u32 *blocked = bpf_map_lookup_elem(&blacklist_map, &src_ip);
    if (blocked)
        return XDP_DROP;

    return XDP_PASS;
}

#!/usr/bin/env python3
"""
FluxGuard Phase 5 — Prometheus text metrics over HTTP (stdlib only).

Exposes GET /metrics on a background thread; state is updated from the brain.
"""

from __future__ import annotations

import threading
from http.server import BaseHTTPRequestHandler, HTTPServer
from typing import Dict


class MetricsState:
    """Thread-safe snapshot for Prometheus exposition."""

    def __init__(self) -> None:
        self._lock = threading.Lock()
        self._packets_by_ip: Dict[str, int] = {}
        self._blocked_ips = 0
        self._block_events = 0
        self._unblock_events = 0

    def update_snapshot(self, packets_by_ip: Dict[str, int], blocked_ips: int) -> None:
        with self._lock:
            self._packets_by_ip = dict(packets_by_ip)
            self._blocked_ips = int(blocked_ips)

    def inc_block_events(self) -> None:
        with self._lock:
            self._block_events += 1

    def inc_unblock_events(self) -> None:
        with self._lock:
            self._unblock_events += 1

    def render_prometheus_text(self) -> str:
        with self._lock:
            packets = sorted(self._packets_by_ip.items())
            blocked = self._blocked_ips
            be = self._block_events
            ue = self._unblock_events

        lines = [
            "# HELP fluxguard_packets_total Current packet counter per source IP from meter_map.",
            "# TYPE fluxguard_packets_total gauge",
        ]
        for ip, cnt in packets:
            safe = ip.replace("\\", "\\\\").replace('"', '\\"')
            lines.append('fluxguard_packets_total{ip="%s"} %d' % (safe, cnt))
        lines.extend(
            [
                "# HELP fluxguard_blocked_ips_total Number of IPs currently blocked by the brain.",
                "# TYPE fluxguard_blocked_ips_total gauge",
                "fluxguard_blocked_ips_total %d" % blocked,
                "# HELP fluxguard_block_events_total Cumulative block events since brain start.",
                "# TYPE fluxguard_block_events_total counter",
                "fluxguard_block_events_total %d" % be,
                "# HELP fluxguard_unblock_events_total Cumulative unblock events since brain start.",
                "# TYPE fluxguard_unblock_events_total counter",
                "fluxguard_unblock_events_total %d" % ue,
                "",
            ]
        )
        return "\n".join(lines)


def start_metrics_server(state: MetricsState, host: str, port: int) -> threading.Thread:
    """Start HTTP /metrics in a daemon thread. Returns the thread object."""

    class Handler(BaseHTTPRequestHandler):
        def log_message(self, fmt: str, *args: object) -> None:
            return

        def do_GET(self) -> None:  # noqa: N802 (stdlib API)
            if self.path != "/metrics":
                self.send_response(404)
                self.end_headers()
                return
            body = state.render_prometheus_text().encode("utf-8")
            self.send_response(200)
            self.send_header("Content-Type", "text/plain; version=0.0.4; charset=utf-8")
            self.send_header("Content-Length", str(len(body)))
            self.end_headers()
            self.wfile.write(body)

    def run() -> None:
        server = HTTPServer((host, port), Handler)
        server.serve_forever()

    thread = threading.Thread(target=run, name="fluxguard-metrics", daemon=True)
    thread.start()
    return thread
EOF_METRICS

#!/usr/bin/env python3
"""
FluxGuard Phase 5 - Brain + metrics + JSON logging

Ubuntu-focused controller that:
1) Reads packet counters from meter_map using bpftool --json
2) Detects abusive packet rates per source IP
3) Writes offenders to blacklist_map
4) Auto-unblocks offenders after cooldown ("amnesty")
5) Exposes Prometheus /metrics (fluxguard_metrics.py) and JSON line logs

Run from host namespace (recommended):
    sudo python3 fluxguard_brain.py --netns fluxguard
"""

from __future__ import annotations

import argparse
import datetime
import ipaddress
import json
import subprocess
import sys
import time
from dataclasses import dataclass
from typing import Any, Dict, List, Optional, Tuple

from fluxguard_metrics import MetricsState, start_metrics_server


@dataclass
class Config:
    netns: str
    meter_map: str
    blacklist_map: str
    pps_threshold: float
    poll_interval: float
    cooldown_sec: int
    block_batch_size: int
    verbose: bool
    log_file: str
    metrics_host: str
    metrics_port: int


def run_cmd(cmd: List[str]) -> str:
    """Run command and return stdout. Raise RuntimeError on failure."""
    try:
        proc = subprocess.run(
            cmd,
            check=True,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )
    except subprocess.CalledProcessError as exc:
        stderr = exc.stderr.strip()
        stdout = exc.stdout.strip()
        msg = (
            f"Command failed: {' '.join(cmd)}\n"
            f"exit={exc.returncode}\n"
            f"stdout={stdout}\n"
            f"stderr={stderr}"
        )
        raise RuntimeError(msg) from exc
    return proc.stdout


def bpftool_json(netns: str, args: List[str]) -> object:
    """
    Execute bpftool in target netns and parse JSON output.
    Always forces --json for cross-version stability.
    """
    cmd = ["sudo", "ip", "netns", "exec", netns, "bpftool", "--json"] + args
    output = run_cmd(cmd)
    try:
        return json.loads(output)
    except json.JSONDecodeError as exc:
        raise RuntimeError(
            f"Invalid JSON from bpftool for args {args}:\n{output}"
        ) from exc


def ip_to_key_hex(ip_str: str) -> str:
    """Convert dotted IPv4 to bpftool key hex format string."""
    ip = ipaddress.IPv4Address(ip_str)
    return " ".join(f"{b:02x}" for b in ip.packed)


def extract_hex_tokens_from_field(field: object) -> List[str]:
    """
    Normalize bpftool JSON key/value field into a list of hex byte tokens.
    Handles several bpftool JSON shape variations.
    """
    tokens: List[str] = []

    if isinstance(field, list):
        for item in field:
            if isinstance(item, int):
                tokens.append(f"{item & 0xFF:02x}")
            elif isinstance(item, str):
                cleaned = item.strip().lower().replace("0x", "")
                if cleaned:
                    tokens.extend(cleaned.split())
    elif isinstance(field, str):
        cleaned = field.strip().lower().replace("0x", "")
        if cleaned:
            tokens.extend(cleaned.split())
    elif isinstance(field, int):
        tokens.append(f"{field & 0xFF:02x}")
    elif isinstance(field, dict):
        # Try common nested forms: {"bytes":[...]} or {"data":".."}
        for nested_key in ("bytes", "data", "raw"):
            if nested_key in field:
                return extract_hex_tokens_from_field(field[nested_key])

    return tokens


def key_tokens_to_ipv4(tokens: List[str]) -> Optional[str]:
    """Convert first 4 tokens to IPv4 dotted string."""
    if len(tokens) < 4:
        return None
    try:
        b = bytes(int(t, 16) for t in tokens[:4])
        return str(ipaddress.IPv4Address(b))
    except Exception:
        return None


def value_tokens_to_u32(tokens: List[str]) -> Optional[int]:
    """
    Parse little-endian u32 counter from first 4 value bytes.
    Expected for typical `__u32` meter values.
    """
    if len(tokens) < 4:
        return None
    try:
        b0, b1, b2, b3 = (int(tokens[i], 16) for i in range(4))
        return b0 | (b1 << 8) | (b2 << 16) | (b3 << 24)
    except Exception:
        return None


def parse_meter_map_dump(raw: object) -> Dict[str, int]:
    """
    Parse bpftool map dump JSON into {ip: packet_counter}.
    Robust against common format differences across versions.
    """
    result: Dict[str, int] = {}

    if not isinstance(raw, list):
        return result

    for entry in raw:
        if not isinstance(entry, dict):
            continue

        key_obj = entry.get("key")
        val_obj = entry.get("value")
        if key_obj is None or val_obj is None:
            continue

        key_tokens = extract_hex_tokens_from_field(key_obj)
        val_tokens = extract_hex_tokens_from_field(val_obj)

        ip = key_tokens_to_ipv4(key_tokens)
        count = value_tokens_to_u32(val_tokens)

        if ip is None or count is None:
            continue
        result[ip] = count

    return result


def append_json_log(path: str, event: str, fields: Dict[str, Any]) -> None:
    """Append one JSON object per line (ISO 8601 timestamp, structured fields)."""
    ts = (
        datetime.datetime.now(datetime.timezone.utc).strftime("%Y-%m-%dT%H:%M:%S.%f")[:-3]
        + "Z"
    )
    record: Dict[str, Any] = {"timestamp": ts, "event": event}
    record.update(fields)
    line = json.dumps(record, separators=(",", ":")) + "\n"
    with open(path, "a", encoding="utf-8") as fh:
        fh.write(line)


def dump_meter_map(cfg: Config) -> Dict[str, int]:
    raw = bpftool_json(cfg.netns, ["map", "dump", "name", cfg.meter_map])
    return parse_meter_map_dump(raw)


def map_update_blacklist(cfg: Config, ip_str: str) -> None:
    key_hex = ip_to_key_hex(ip_str)
    # Value is little-endian u32=1 -> 01 00 00 00
    cmd = [
        "sudo",
        "ip",
        "netns",
        "exec",
        cfg.netns,
        "bpftool",
        "map",
        "update",
        "name",
        cfg.blacklist_map,
        "key",
        "hex",
    ] + key_hex.split() + ["value", "hex", "01", "00", "00", "00"]
    run_cmd(cmd)


def map_delete_blacklist(cfg: Config, ip_str: str) -> None:
    key_hex = ip_to_key_hex(ip_str)
    cmd = [
        "sudo",
        "ip",
        "netns",
        "exec",
        cfg.netns,
        "bpftool",
        "map",
        "delete",
        "name",
        cfg.blacklist_map,
        "key",
        "hex",
    ] + key_hex.split()
    run_cmd(cmd)


def monitor(cfg: Config, metrics: MetricsState) -> None:
    prev_counts: Dict[str, int] = {}
    prev_ts = time.monotonic()
    blocked_until: Dict[str, float] = {}

    print(
        "FluxGuard brain started "
        f"(threshold={cfg.pps_threshold:.1f} pps, cooldown={cfg.cooldown_sec}s)"
    )
    print(
        "Metrics: http://%s:%d/metrics  Log: %s"
        % (cfg.metrics_host, cfg.metrics_port, cfg.log_file)
    )
    print("Press Ctrl+C to stop.")

    while True:
        now = time.monotonic()
        dt = now - prev_ts
        if dt <= 0:
            dt = cfg.poll_interval

        try:
            curr_counts = dump_meter_map(cfg)
        except RuntimeError as exc:
            print(f"[WARN] meter_map read failed: {exc}", file=sys.stderr)
            time.sleep(cfg.poll_interval)
            prev_ts = time.monotonic()
            continue

        # Detect offenders and batch block.
        to_block: List[Tuple[str, float, int]] = []
        for ip, curr in curr_counts.items():
            prev = prev_counts.get(ip, curr)

            # Handle map resets / counter wrap gracefully.
            delta = curr - prev if curr >= prev else curr
            pps = delta / dt

            if pps >= cfg.pps_threshold and ip not in blocked_until:
                to_block.append((ip, pps, curr))

        if to_block:
            # Highest PPS first so severe offenders get priority.
            to_block.sort(key=lambda x: x[1], reverse=True)
            if cfg.block_batch_size > 0:
                to_block = to_block[: cfg.block_batch_size]

            for ip, pps, counter in to_block:
                try:
                    map_update_blacklist(cfg, ip)
                    blocked_until[ip] = now + cfg.cooldown_sec
                    metrics.inc_block_events()
                    print(
                        f"[BLOCK] ip={ip} pps={pps:.1f} "
                        f"counter={counter} unblock_in={cfg.cooldown_sec}s"
                    )
                    append_json_log(
                        cfg.log_file,
                        "BLOCK",
                        {
                            "ip": ip,
                            "pps": round(pps, 6),
                            "counter": counter,
                            "blocked_count": len(blocked_until),
                            "dt": round(dt, 6),
                        },
                    )
                except RuntimeError as exc:
                    print(f"[WARN] block failed for {ip}: {exc}", file=sys.stderr)

        # Amnesty: auto-unblock after cooldown.
        if blocked_until:
            expired = [ip for ip, ts in blocked_until.items() if now >= ts]
            for ip in expired:
                try:
                    map_delete_blacklist(cfg, ip)
                    metrics.inc_unblock_events()
                    print(f"[UNBLOCK] ip={ip} cooldown_expired")
                    append_json_log(
                        cfg.log_file,
                        "UNBLOCK",
                        {
                            "ip": ip,
                            "blocked_count": len(blocked_until) - 1,
                            "dt": round(dt, 6),
                        },
                    )
                except RuntimeError as exc:
                    # If key already removed, continue to avoid lock-in.
                    print(f"[WARN] unblock failed for {ip}: {exc}", file=sys.stderr)
                finally:
                    blocked_until.pop(ip, None)

        active_blocked = len(blocked_until)
        observed = len(curr_counts)
        append_json_log(
            cfg.log_file,
            "TICK",
            {
                "observed_ips": observed,
                "blocked_count": active_blocked,
                "dt": round(dt, 6),
            },
        )
        metrics.update_snapshot(curr_counts, active_blocked)

        if cfg.verbose:
            print(
                f"[TICK] observed_ips={observed} blocked_ips={active_blocked} dt={dt:.3f}s"
            )

        prev_counts = curr_counts
        prev_ts = now
        time.sleep(cfg.poll_interval)
